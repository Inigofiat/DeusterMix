160 // Método para obtener la lista de IngredienteDTO a partir de una lista de IDs
161 private List<IngredienteDTO> obtenerIngredientesDTOPorIds(List<Long> idIngredientes) {
162 List<IngredienteDTO> ingredientesDTO = new ArrayList<>();
163 if (idIngredientes != null) {
164 for (Long idIngrediente : idIngredientes) {
165 String nombre = servicioReceta.getNombreIngrediente(idIngrediente);
166 if (nombre != null) {

174 private List<RecetaDTO> recetasARecetaDTO(List<Receta> recetas) {

183 private RecetaDTO recetaARecetaDTO(Receta receta) {
184 List<Long> idIngredientes = new ArrayList<>();
185 if (receta.getIngredientes() != null) {
186 for (Ingrediente ingrediente : receta.getIngredientes()) {
187 idIngredientes.add(ingrediente.getId());
188 }
189 }
190
191 // Crear un DTO reducido del cliente para evitar referencias circulares
192 Cliente clienteOriginal = receta.getCliente();
193 ClienteReducidoDTO clienteDTO = null;
194 if (clienteOriginal != null) {
195 clienteDTO = new ClienteReducidoDTO(
196 clienteOriginal.getEmail(),
197 clienteOriginal.getNombre() // Asumiendo que Cliente tiene un método getNombre()
198 );
199 }
200
201 RecetaDTO recetaDTO = new RecetaDTO(
202 receta.getId(),
203 receta.getNombre(),
204 receta.getDescripcion(),
205 receta.getInstrucciones(),
206 receta.getImageUrl(),
207 idIngredientes,
208 clienteDTO
209 );
210 return recetaDTO;
